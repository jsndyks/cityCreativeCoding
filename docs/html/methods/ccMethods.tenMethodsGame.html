Solution - Catch up Here! 
long fibonacciR(int n) {

  if (n <= 0) {
    return 0;   // -- base case
  } else if (n == 1) {
    return 1;   // -- base case
  } else {
    return fibonacciR(n - 1) + fibonacciR(n - 2); // -- recursive case
  }
}
