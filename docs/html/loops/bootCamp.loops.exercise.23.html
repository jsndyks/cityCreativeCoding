3. Copy the code from the lecture notes that draws a line of trees (reproduced below) and adapt it so that it draws three lines of trees. 

color leafColour,trunkColour;

void setup()
{
  size(800, 400);
  noStroke();
  leafColour  = color(82, 144, 97);
  trunkColour = color(95, 57, 26);
}

void draw()
{
  background(157, 238, 247);   // Sky
  fill(156, 245, 170);         // Ground.
  rect(0, height/2, width, height/2);

  for (float xPos=50; xPos<width; xPos=xPos+100)
  {
    drawTree(xPos,180);
  }
}

// Draws a single tree at the given position.
void drawTree(float treeX, float treeY)
{
  fill(trunkColour);
  rect(treeX, treeY, 20, 70);
  fill(leafColour);
  ellipse(treeX+10, treeY-30, 70, 90);
}

TIP Calculating whether a point is inside or outside a non-circular ellipse is a little more complicated than for a circle. But we can use our own method do the calculation: 

// Will return true if position (x,y) is within an ellipse(cx,cy,w,h), otherwise false.
boolean isInEllipse(float x, float y, float cx, float cy, float w, float h) 
{
  return (x-cx)*(x-cx)/(w*w/4) + (y-cy)*(y-cy)/(h*h/4) < 1;
}
