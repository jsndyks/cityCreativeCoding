Op art or 'optical art' is a form of abstract art that emerged as a movement in the 1960s. It explores the differences between what we see and what we understand when looking at an image. It typically involves the use of black and white or contrasting colours to create visual illusions, often generating a sense of movement even when none exists. It provides a good topic for some Processing sketches because it is often made up of a set of repeated simple shapes - something a programming language is good at reproducing.

Try doing a Google image search for op art to get an idea of the types of op art images that can be produced. Think about how you might recreate some of these images by getting Processing to draw multiple copies of simple shapes in contrasting colours. There is no need to reproduce such images exactly, but they should provide some inspiration for your own designs.

Some project ideas:

  • Choose a selection of op art designs and try to reproduce something similar with Processing.
  • You could try experimenting with different shapes or transformations.
  • How might you incorporate animation into your sketch?
  • How might you use interaction via the mouse or keyboard to enhance your sketch?

After Movement in Squares by Brigit Riley 1961.

void setup()
{
  size(600, 400);
  noLoop();
}

void draw()
{
  background(255);
  noStroke();
  fill(0);

  float maxSquareSize = 40;  // Height and maximum width of each rectangle.
  float rectWidth = maxSquareSize;
  float widthChange = -2;    // Amount by which rectangle shrinks in each column
  float xPos = 0;
  float yOffset = 0;

  while (xPos < width)
  {
    for (float yPos=0; yPos<height; yPos+= 2*maxSquareSize)
    {
      rect(xPos, yPos+yOffset, rectWidth, maxSquareSize);
    }

    xPos      += rectWidth;  // Move to next column,
    rectWidth += widthChange;  // Change width of rectangles.

    // Alternate each column by offsetting y position every other column.
    if (yOffset == 0)
    {
      yOffset = maxSquareSize;
    }
    else
    {
      yOffset = 0;
    }

    // If width has decreased to less than 4, start increasing it.
    if (rectWidth < 4)
    {
      widthChange = 2;
      rectWidth  = 4;
    }
  }
}

void setup()
{
  size(800, 400);
  noLoop();
}

void draw()
{
  background(255);
  noStroke();
  rectMode(CENTER);    // Square positioned by centre coordinates.
  boolean isBlack = true;

  float squareSize = width;
  while (squareSize > 12)
  {
    // Alternate between black and white.
    if (isBlack)
    {
      fill(0);
      isBlack = false;
    }
    else
    {
      fill(255);
      isBlack = true;
    }

    // Rotate square by 5 degrees about its centre.
    translate(width/2, height/2);
    rotate(radians(5));
    translate(-width/2, -height/2);

    rect(width/2, height/2, squareSize, squareSize);

    // Shrink size of square a little
    squareSize -= 16;
  }
}

void draw()
{
  background(255);
  noStroke();
 
  boolean isBlack = true;
  
  float circleSize = 340;
  while (circleSize > 12)
  {
    // Alternate between black and white.
    if (isBlack)
    {
      fill(0);
      isBlack = false;
    }
    else
    {
      fill(255);
      isBlack = true;
    }
    
    // Create small random offset for each circle.
    float xOffset = noise(circleSize*0.1+frameCount*0.005)*30;
    float yOffset = noise(circleSize*0.1+frameCount*0.005,1)*30;
    
    ellipse(170+xOffset,height/2+yOffset,circleSize,circleSize);
    
    // Shrink size of circle a little
    circleSize -= 16;
  }
}
